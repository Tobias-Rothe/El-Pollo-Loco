/**
 * Manages the sound state based on the world's mute status.
 *
 * @function manageSounds
 * @memberof WorldSounds
 * @param {Object} world - The world object containing the mute status.
 * @param {Object} sound - The sound object to be managed.
 * @property {boolean} world.isMuted - Indicates if the world is muted.
 * @property {boolean} sound.muted - Indicates if the sound is muted.
 * @property {boolean} sound.wasPlaying - Indicates if the sound was playing before being paused.
 * @property {boolean} sound.paused - Indicates if the sound is paused.
 * @property {number} sound.currentTime - The current playback time of the sound.
 */

/**
 * Stops all sounds in the world.
 *
 * @function stopAllSounds
 * @memberof WorldSounds
 * @param {Object} world - The world object containing the sounds.
 * @property {Function} world.getAllGameSounds - A function that returns all game sounds.
 */

/**
 * Toggles the mute status of the world and manages all sounds accordingly.
 *
 * @function toggleMute
 * @memberof WorldSounds
 * @param {Object} world - The world object containing the mute status and sounds.
 * @property {boolean} world.isMuted - Indicates if the world is muted.
 * @property {Function} world.getAllGameSounds - A function that returns all game sounds.
 */
const WorldSounds = {
  manageSounds(world, sound) {
    if (sound) {
      sound.muted = world.isMuted;
      if (world.isMuted) {
        sound.pause();
        sound.currentTime = 0;
        sound.wasPlaying = !sound.paused;
      } else if (sound.wasPlaying) {
        sound.play();
      }
    }
  },

  stopAllSounds(world) {
    world.getAllGameSounds().forEach((sound) => {
      sound.pause();
      sound.currentTime = 0;
      sound.wasPlaying = false;
    });
  },

  toggleMute(world) {
    world.isMuted = !world.isMuted;
    world.getAllGameSounds().forEach((sound) => this.manageSounds(world, sound));
  },
};
